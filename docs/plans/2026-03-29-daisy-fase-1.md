# Daisy Fase 1 — Implementatieplan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** TC-kern kan chatten met Daisy in de Beheer-app en informatie opvragen over planning, leden en teams.

**Architecture:** Eén streaming API route (`/api/ai/chat`) met Vercel AI SDK + Gemini Flash. ChatPanel als Drawer in de Beheer-layout. Drie read-only plugins (planning, monitor, teamindeling) als tools die Gemini kan aanroepen. Gesprekgeschiedenis in PostgreSQL.

**Tech Stack:** Vercel AI SDK v4, `@ai-sdk/google` (Gemini 2.0 Flash), Prisma, React 19, Framer Motion, Zod

**Spec:** `docs/specs/2026-03-29-daisy-design.md`

```typescript
// apps/web/src/lib/ai/plugins/monitor.ts
import { tool } from "ai";
import { z } from "zod";
import { prisma } from "@/lib/db/prisma";
import { HUIDIG_SEIZOEN } from "@oranje-wit/types";

export const monitorTools = {
  ledenPerCategorie: tool({
    description: "Geeft het aantal actieve leden per leeftijdscategorie (kleur) voor het huidige seizoen. Gebruik bij vragen over ledenaantallen.",
    parameters: z.object({}),
    execute: async () => {
      const teams = await prisma.oWTeam.findMany({
        where: { seizoen: HUIDIG_SEIZOEN, kleur: { not: null } },
        select: { kleur: true, owCode: true },
      });
      const tellingen = await prisma.competitieSpeler.groupBy({
        by: ["team"],
        where: { seizoen: HUIDIG_SEIZOEN, team: { in: teams.map((t) => t.owCode) } },
        _count: { relCode: true },
      });
      const telMap = new Map(tellingen.map((t) => [t.team, t._count.relCode]));
      const perKleur = new Map<string, number>();
      for (const team of teams) {
        if (!team.kleur) continue;
        perKleur.set(team.kleur, (perKleur.get(team.kleur) ?? 0) + (telMap.get(team.owCode) ?? 0));
      }
      return Array.from(perKleur.entries()).map(([kleur, aantal]) => ({ kleur, aantal })).sort((a, b) => a.kleur.localeCompare(b.kleur));
    },
  }),

  teamBezetting: tool({
    description: "Geeft de bezetting (aantal spelers) per team voor het huidige seizoen. Gebruik bij vragen over teams en hun grootte.",
    parameters: z.object({
      kleur: z.string().optional().describe("Filter op kleur/leeftijdsgroep (bijv. 'geel', 'rood'). Leeg = alle teams."),
    }),
    execute: async ({ kleur }) => {
      const teams = await prisma.oWTeam.findMany({
        where: {
          seizoen: HUIDIG_SEIZOEN,
          naam: { not: null },
          ...(kleur ? { kleur: { equals: kleur, mode: "insensitive" } } : {}),
        },
        select: {
          naam: true,
          kleur: true,
          owCode: true,
        },
        orderBy: [{ kleur: "asc" }, { naam: "asc" }],
      });

      const tellingen = await prisma.competitieSpeler.groupBy({
        by: ["team"],
        where: {
          seizoen: HUIDIG_SEIZOEN,
          team: { in: teams.map((t) => t.owCode) },
        },
        _count: { relCode: true },
      });

      const telMap = new Map(tellingen.map((t) => [t.team, t._count.relCode]));
      return teams.map((t) => ({
        team: t.naam,
        kleur: t.kleur,
        spelers: telMap.get(t.owCode) ?? 0,
      }));
    },
  }),
```

```typescript
// apps/web/src/lib/ai/plugins/teamindeling.ts
import { tool } from "ai";
import { z } from "zod";
import { prisma } from "@/lib/db/prisma";
import { HUIDIG_SEIZOEN } from "@oranje-wit/types";

export const teamindelingTools = {
  spelersInTeam: tool({
    description: "Geeft de spelers in een specifiek team. Gebruik bij vragen als 'wie zit er in team X?'",
    parameters: z.object({
      teamNaam: z.string().describe("Naam van het team (bijv. 'U15-1', 'Senioren 2', 'Geel-3')"),
    }),
    execute: async ({ teamNaam }) => {
      const team = await prisma.oWTeam.findFirst({
        where: {
          seizoen: HUIDIG_SEIZOEN,
          naam: { contains: teamNaam, mode: "insensitive" },
        },
      });
      if (!team) return { error: `Team "${teamNaam}" niet gevonden in ${HUIDIG_SEIZOEN}.` };

      const spelers = await prisma.competitieSpeler.findMany({
        where: { seizoen: HUIDIG_SEIZOEN, team: team.owCode },
        select: { relCode: true, naam: true, geslacht: true },
        orderBy: { naam: "asc" },
      });

      return {
        team: team.naam,
        kleur: team.kleur,
        categorie: team.categorie,
        spelers: spelers.map((s) => ({
          naam: s.naam,
          geslacht: s.geslacht,
        })),
        totaal: spelers.length,
      };
    },
  }),

  werkindelingStatus: tool({
    description: "Geeft de status van de werkindeling voor het volgende seizoen. Gebruik bij vragen over de teamindeling.",
    parameters: z.object({}),
    execute: async () => {
      const blauwdruk = await prisma.blauwdruk.findFirst({
        orderBy: { createdAt: "desc" },
        select: {
          id: true,
          seizoen: true,
          status: true,
          createdAt: true,
          _count: { select: { concepten: true } },
        },
      });
      if (!blauwdruk) return { status: "Geen blauwdruk gevonden. Het teamindelingsproces is nog niet gestart." };

      const scenarios = await prisma.scenario.findMany({
        where: { concept: { blauwdrukId: blauwdruk.id } },
        select: {
          naam: true,
          status: true,
          isWerkindeling: true,
          _count: { select: { teams: true } },
        },
      });

      return {
        seizoen: blauwdruk.seizoen,
        blauwdrukStatus: blauwdruk.status,
        aantalConcepten: blauwdruk._count.concepten,
        scenarios: scenarios.map((s) => ({
          naam: s.naam,
          status: s.status,
          isWerkindeling: s.isWerkindeling,
          aantalTeams: s._count.teams,
        })),
      };
    },
  }),
```

```typescript
// apps/web/src/lib/ai/daisy.ts
import type { AuthSession } from "@oranje-wit/auth/checks";
import { HUIDIG_SEIZOEN } from "@oranje-wit/types";

const MAANDEN = [
  "januari", "februari", "maart", "april", "mei", "juni",
  "juli", "augustus", "september", "oktober", "november", "december",
];

function seizoensPeriode(): string {
  const maand = new Date().getMonth(); // 0-11
  if (maand >= 7 && maand <= 8) return "Start (aug-sep): vierkante wielen, alles nieuw";
  if (maand >= 9 || maand <= 1) return "Draaiend (okt-feb): competitie, opleidingen";
  return "Oogsten & Zaaien (mrt-jun): evaluaties, teamindeling volgend seizoen";
}

export function buildDaisyPrompt(session: AuthSession): string {
  const nu = new Date();
  const maand = MAANDEN[nu.getMonth()];

  return `Je bent Daisy, het 4e lid van de kern-TC van c.k.v. Oranje Wit.
Daisy staat voor: Doet Alle Irritante Shit, Yo!

Je praat met ${session.user.name || session.user.email}.
Rol: ${session.user.isTC ? "TC-lid" : "Gebruiker"}, clearance ${session.user.clearance}.
Seizoen: ${HUIDIG_SEIZOEN}. Vandaag: ${nu.toLocaleDateString("nl-NL")} (${maand}).
Seizoensperiode: ${seizoensPeriode()}.

## Gedrag
- Antwoord altijd in het Nederlands, informeel en direct.
- Noem de vereniging "c.k.v. Oranje Wit" of "OW".
- Wees bondig. Geen lange inleidingen.
- Gebruik de beschikbare tools om data op te halen. Verzin geen data.
- Als je iets niet weet of niet kunt, zeg dat eerlijk.

## De Oranje Draad
Het technisch beleid: PLEZIER + ONTWIKKELING + PRESTATIE → DUURZAAMHEID.
- Plezier is altijd hoogste prioriteit, niet onderhandelbaar.
- Ontwikkeling is het echte doel.
- Prestatie is een middel, nooit einddoel.

## Jouw rol
Je helpt de TC met informatie opvragen, planning overzien, en data analyseren.
Je neemt geen besluiten — dat doet de TC. Je levert de informatie aan.`;
}
```

```typescript
import { prisma } from "@/lib/db/prisma";
import { HUIDIG_SEIZOEN } from "@oranje-wit/types";
import type { BerichtRol } from "@oranje-wit/database/generated/prisma";

export async function getOfMaakGesprek(userId: string, gesprekId?: string) {
  if (gesprekId) {
    const bestaand = await prisma.aiGesprek.findFirst({
      where: { id: gesprekId, userId },
    });
    if (bestaand) return bestaand;
  }

  return prisma.aiGesprek.create({
    data: { userId, seizoen: HUIDIG_SEIZOEN },
  });
}

export async function slaBerichtOp(
  gesprekId: string,
  rol: BerichtRol,
  inhoud: string,
  metadata?: Record<string, unknown>,
) {
  return prisma.aiBericht.create({
    data: { gesprekId, rol, inhoud, metadata: metadata ?? undefined },
  });
}

export async function getGesprekken(userId: string) {
  return prisma.aiGesprek.findMany({
    where: { userId, seizoen: HUIDIG_SEIZOEN },
    orderBy: { updatedAt: "desc" },
    take: 20,
    select: {
      id: true,
      titel: true,
      updatedAt: true,
    },
  });
}

export async function getBerichten(gesprekId: string, userId: string) {
  const gesprek = await prisma.aiGesprek.findFirst({
    where: { id: gesprekId, userId },
  });
  if (!gesprek) return [];

  return prisma.aiBericht.findMany({
    where: { gesprekId },
    orderBy: { createdAt: "asc" },
    select: { id: true, rol: true, inhoud: true, createdAt: true },
  });
}
```

```typescript
// apps/web/src/app/api/ai/chat/route.ts
import { streamText } from "ai";
import { google } from "@ai-sdk/google";
import { guardAuth } from "@oranje-wit/auth/checks";
import { fail } from "@/lib/api/response";
import { buildDaisyPrompt } from "@/lib/ai/daisy";
import { getDaisyTools } from "@/lib/ai/plugins/registry";
import { getOfMaakGesprek, slaBerichtOp } from "@/lib/ai/gesprekken";
import type { Clearance } from "@oranje-wit/types";

export const maxDuration = 30;

export async function POST(request: Request) {
  const auth = await guardAuth();
  if (!auth.ok) return auth.response;

  const { session } = auth;
  const clearance = (session.user.clearance ?? 0) as Clearance;

  if (clearance < 1) {
    return fail("Daisy is beschikbaar voor TC-leden en coördinatoren.", 403);
  }

  const body = await request.json();
  const { messages, gesprekId } = body;

  if (!messages || !Array.isArray(messages) || messages.length === 0) {
    return fail("Geen berichten meegegeven.", 400);
  }

  // Gesprek opslaan
  const gesprek = await getOfMaakGesprek(session.user.email!, gesprekId);

  // Laatste user-bericht opslaan
  const laatsteBericht = messages[messages.length - 1];
  if (laatsteBericht.role === "user") {
    await slaBerichtOp(gesprek.id, "GEBRUIKER", laatsteBericht.content);
  }

  const systemPrompt = buildDaisyPrompt(session);
  const tools = getDaisyTools(clearance);

  const result = streamText({
    model: google("gemini-2.0-flash"),
    system: systemPrompt,
    messages,
    tools,
    maxSteps: 3,
    onFinish: async ({ text }) => {
      if (text) {
        await slaBerichtOp(gesprek.id, "ASSISTENT", text);
      }
      // Titel genereren bij eerste bericht
      if (!gesprek.titel && messages.length <= 2) {
        const { prisma } = await import("@/lib/db/prisma");
        await prisma.aiGesprek.update({
          where: { id: gesprek.id },
          data: { titel: laatsteBericht.content.slice(0, 80) },
        });
      }
    },
  });

  return result.toDataStreamResponse({
    headers: { "X-Gesprek-Id": gesprek.id },
  });
}
```

```tsx
  onChange: (e: React.ChangeEvent<HTMLTextAreaElement>) => void;
  onSubmit: (e: FormEvent) => void;
  isLoading: boolean;

export function ChatInput({ input, onChange, onSubmit, isLoading }: ChatInputProps) {
  const ref = useRef<HTMLTextAreaElement>(null);

  function handleKeyDown(e: React.KeyboardEvent) {
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      if (input.trim() && !isLoading) {
        onSubmit(e as unknown as FormEvent);
      }
    }
  }

  return (
    <form onSubmit={onSubmit} className="flex items-end gap-2 p-3" style={{ borderTop: "1px solid var(--border-default)" }}>
      <textarea
        ref={ref}
        value={input}
        onChange={onChange}
        onKeyDown={handleKeyDown}
        placeholder="Vraag Daisy..."
        rows={1}
        disabled={isLoading}
        className="flex-1 resize-none rounded-lg px-3 py-2 text-sm outline-none placeholder:opacity-50"
        style={{
          backgroundColor: "var(--surface-sunken)",
          color: "var(--text-primary)",
          border: "1px solid var(--border-default)",
          minHeight: "40px",
          maxHeight: "120px",
        }}
      />
      <button
        type="submit"
        disabled={!input.trim() || isLoading}
        className="flex h-10 w-10 shrink-0 items-center justify-center rounded-lg transition-opacity disabled:opacity-30"
        style={{ backgroundColor: "var(--ow-oranje-500)", color: "white" }}
      >
        <svg className="h-4 w-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M5 12h14M12 5l7 7-7 7" />
        </svg>
      </button>
    </form>
  );
}
```

```tsx
// apps/web/src/components/daisy/chat-panel.tsx
"use client";

import { useRef, useEffect } from "react";
import { useChat } from "ai/react";
import { motion, AnimatePresence } from "framer-motion";
import { drawerSlide, overlayBackdrop } from "@oranje-wit/ui/motion/variants";
import { ChatBericht } from "./chat-bericht";
import { ChatInput } from "./chat-input";

interface ChatPanelProps {
  open: boolean;
  onClose: () => void;
}

export function ChatPanel({ open, onClose }: ChatPanelProps) {
  const scrollRef = useRef<HTMLDivElement>(null);
  const closeRef = useRef<HTMLButtonElement>(null);

  const { messages, input, handleInputChange, handleSubmit, isLoading, error } =
    useChat({ api: "/api/ai/chat" });

  // Scroll naar beneden bij nieuwe berichten
  useEffect(() => {
    if (scrollRef.current) {
      scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
    }
  }, [messages]);

  // Escape + focus + scroll lock
  useEffect(() => {
    if (!open) return;
    requestAnimationFrame(() => closeRef.current?.focus());

    const handleEscape = (e: KeyboardEvent) => {
      if (e.key === "Escape") onClose();
    };
    document.addEventListener("keydown", handleEscape);
    const orig = document.body.style.overflow;
    document.body.style.overflow = "hidden";

    return () => {
      document.removeEventListener("keydown", handleEscape);
      document.body.style.overflow = orig;
    };
  }, [open, onClose]);

  const variants = drawerSlide("right");

  return (
    <AnimatePresence>
      {open && (
        <div className="fixed inset-0 z-[80]">
          {/* Backdrop */}
          <motion.div
            variants={overlayBackdrop}
            initial="hidden"
            animate="visible"
            exit="exit"
            className="absolute inset-0"
            style={{ backgroundColor: "var(--surface-scrim)" }}
            onClick={onClose}
          />

          {/* Panel */}
          <motion.div
            variants={variants}
            initial="hidden"
            animate="visible"
            exit="exit"
            className="absolute top-0 right-0 flex h-full w-full flex-col sm:w-[380px]"
            style={{
              backgroundColor: "rgba(26, 29, 35, 0.95)",
              backdropFilter: "blur(20px)",
              WebkitBackdropFilter: "blur(20px)",
              borderLeft: "1px solid var(--border-default)",
              boxShadow: "var(--shadow-modal)",
            }}
            role="dialog"
            aria-modal="true"
            aria-label="Daisy chat"
          >
            {/* Header */}
            <div
              className="flex shrink-0 items-center justify-between px-4 py-3"
              style={{ borderBottom: "1px solid var(--border-default)" }}
            >
              <div className="flex items-center gap-2">
                <span
                  className="flex h-7 w-7 items-center justify-center rounded-full text-xs font-bold"
                  style={{ backgroundColor: "var(--ow-oranje-500)", color: "white" }}
                >
                  D
                </span>
                <div>
                  <h2 className="text-sm font-bold" style={{ color: "var(--text-primary)" }}>
                    Daisy
                  </h2>
                  <p className="text-[10px]" style={{ color: "var(--text-tertiary)" }}>
                    TC-lid &middot; {isLoading ? "denkt na..." : "online"}
                  </p>
                </div>
              </div>
              <motion.button
                ref={closeRef}
                whileTap={{ scale: 0.9 }}
                onClick={onClose}
                className="flex h-8 w-8 items-center justify-center rounded-lg transition-colors"
                style={{ color: "var(--text-tertiary)" }}
                aria-label="Sluiten"
              >
                <svg className="h-5 w-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                  <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M6 18L18 6M6 6l12 12" />
                </svg>
              </motion.button>
            </div>

            {/* Berichten */}
            <div ref={scrollRef} className="flex-1 overflow-y-auto px-4 py-4">
              {messages.length === 0 && (
                <div className="flex h-full flex-col items-center justify-center text-center">
                  <span
                    className="mb-3 flex h-12 w-12 items-center justify-center rounded-full text-lg font-bold"
                    style={{ backgroundColor: "var(--ow-oranje-500)", color: "white" }}
                  >
                    D
                  </span>
                  <p className="text-sm font-medium" style={{ color: "var(--text-primary)" }}>
                    Hoi! Ik ben Daisy.
                  </p>
                  <p className="mt-1 text-xs" style={{ color: "var(--text-tertiary)" }}>
                    Vraag me iets over de planning, teams of leden.
                  </p>
                </div>
              )}
              {messages.map((m) => (
                <ChatBericht key={m.id} bericht={m} />
              ))}
              {isLoading && messages[messages.length - 1]?.role === "user" && (
                <div className="mb-3 flex justify-start">
                  <div
                    className="rounded-xl rounded-bl-sm px-3.5 py-2.5 text-sm"
                    style={{ backgroundColor: "var(--surface-raised)", color: "var(--text-tertiary)" }}
                  >
                    ...
                  </div>
                </div>
              )}
              {error && (
                <div className="mb-3 rounded-lg px-3 py-2 text-xs" style={{ backgroundColor: "var(--surface-raised)", color: "#ef4444" }}>
                  Fout: {error.message}
                </div>
              )}
            </div>

            {/* Input */}
            <ChatInput
              input={input}
              onChange={handleInputChange}
              onSubmit={handleSubmit}
              isLoading={isLoading}
            />
          </motion.div>
        </div>
      )}
    </AnimatePresence>
  );
}
```
